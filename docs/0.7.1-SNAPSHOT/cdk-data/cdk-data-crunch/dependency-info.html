Apache Maven

  <artifactId>cdk-data-crunch</artifactId>
  <version>0.7.1-SNAPSHOT</version>
</dependency>

Apache Buildr

'com.cloudera.cdk:cdk-data-crunch:jar:0.7.1-SNAPSHOT'

Apache Ant

</dependency>

Groovy Grape
